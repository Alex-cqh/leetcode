题目描述：
    给你单链表的头节点 head ，请你反转链表，并返回反转后的链表。

//递归
时间复杂度O(n)  空间复杂度O(n)
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if(head==nullptr||head->next==nullptr)
            return head;
        
        ListNode *p=reverseList(head->next);
        //ps:递归函数,写在递归函数后面说明是归的过程中需要进行的操作;
        //            写在递归函数前面说明是递的过程中需要进行的操作.
        head->next->next=head;
        head->next=nullptr;
        return p;
    }
};

//迭代――利用辅助指针
时间复杂度O(n)  空间复杂度O(1)
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        ListNode *next,*pre=nullptr;
        ListNode *cur=head;
        while(cur!=nullptr){
            next=cur->next;
            cur->next=pre;
            pre=cur;
            cur=next;
        }
        return pre;
    }
};
